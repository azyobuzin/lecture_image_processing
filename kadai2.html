def conv_level(img, n):
    # 
    new_img = np.zeros(img.shape, dtype=np.uint)
    
    # 
    # 256
    split_points = np.linspace(0, 256, n, endpoint=False)[1:]
    print(":", split_points)
    
    # 
    for s in split_points:
        # s
        new_img += (img >= s).astype(np.uint8)
        
    return new_img
